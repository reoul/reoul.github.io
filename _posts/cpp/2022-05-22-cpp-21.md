#include <string>

using namespace std;

class A
{
public :
    const char* GetName() const
    {
    	return mName;
    }

    {
    	strcpy(mName, str);
    }

};

{

    void ChangeName()
    {
    	SetName("BBBB");
    }
};

{
public :
    void ChangeName()
    {
    	SetName("CCCC");
    }
};

{
};

{
    D d;
    d.SetName("DDDD");

    C& c = d;

    cout << c.GetName() << endl;
    cout << d.GetName() << endl;

    cout << c.GetName() << endl;
    cout << d.GetName() << endl;
    c.ChangeName();
    cout << b.GetName() << endl;
    cout << c.GetName() << endl;
    cout << d.GetName() << endl;
    return 0;
}
